import os
import rasterio
from rasterio.windows import Window
from tqdm import tqdm

# ========== 配置参数 ==========
data_root = "/mnt/htzzb2/EO_test/datasets/BRIGHT"
input_folders = ["pre-event-mapped", "post-event-mapped", "target"]
output_folders = ["pre-event_slices", "post-event_slices", "target_slices"]
win_size = 128    # 切片窗口大小
stride = 126      # 滑动步长
# ==============================

# 批量创建输出目录
for folder in output_folders:
    os.makedirs(os.path.join(data_root, folder), exist_ok=True)

def slice_tif(input_path, output_dir, win_size, stride):
    """单张影像滑窗切片，保留地理信息，文件名带原图坐标"""
    with rasterio.open(input_path) as src:
        height, width = src.height, src.width
        profile = src.profile

        # 计算行、列方向的窗口起始坐标，自动补全边缘
        row_starts = list(range(0, height - win_size + 1, stride))
        if row_starts[-1] + win_size < height:
            row_starts.append(height - win_size)
        
        col_starts = list(range(0, width - win_size + 1, stride))
        if col_starts[-1] + win_size < width:
            col_starts.append(width - win_size)

        file_stem = os.path.splitext(os.path.basename(input_path))[0]
        # 遍历所有窗口切片
        for row_off in row_starts:
            for col_off in col_starts:
                # 构造读取窗口，计算对应地理变换
                window = Window(col_off, row_off, win_size, win_size)
                win_data = src.read(window=window)
                win_transform = src.window_transform(window)

                # 更新输出配置
                out_profile = profile.copy()
                out_profile.update(
                    height=win_size,
                    width=win_size,
                    transform=win_transform
                )

                # 文件名加入原图坐标：x=列偏移, y=行偏移
                out_name = f"{file_stem}_x{col_off}_y{row_off}.tif"
                out_path = os.path.join(output_dir, out_name)
                with rasterio.open(out_path, 'w', **out_profile) as dst:
                    dst.write(win_data)

# 以第一个文件夹为基准获取所有tif，确保三文件夹文件一一对应
base_dir = os.path.join(data_root, input_folders[0])
tif_list = [f for f in os.listdir(base_dir) if f.lower().endswith('.tif')]

# 批量处理，带进度条
for tif_name in tqdm(tif_list, desc="影像切片进度"):
    for in_dir, out_dir in zip(input_folders, output_folders):
        in_path = os.path.join(data_root, in_dir, tif_name)
        if os.path.isfile(in_path):
            slice_tif(in_path, os.path.join(data_root, out_dir), win_size, stride)

print("所有文件夹滑窗切片完成！")
