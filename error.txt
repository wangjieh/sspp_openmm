import argparse
import os
from concurrent.futures import ProcessPoolExecutor, as_completed
from contextlib import ExitStack

from tqdm import tqdm


DEFAULT_INPUT_FOLDERS = ["pre-event-mapped", "post-event-mapped", "target"]
DEFAULT_OUTPUT_FOLDERS = ["pre-event_slices", "post-event_slices", "target_slices"]


def parse_args():
    parser = argparse.ArgumentParser(
        description="Slice aligned BRIGHT GeoTIFF folders with multiprocessing.")
    parser.add_argument(
        "--data-root",
        default="/mnt/htzzb2/EO_test/datasets/BRIGHT",
        help="Dataset root that contains the input folders.")
    parser.add_argument(
        "--input-folders",
        nargs="+",
        default=DEFAULT_INPUT_FOLDERS,
        help="Input folders under data-root, processed as aligned modalities.")
    parser.add_argument(
        "--output-folders",
        nargs="+",
        default=DEFAULT_OUTPUT_FOLDERS,
        help="Output folders under data-root. Must match input folder count.")
    parser.add_argument(
        "--win-size",
        type=int,
        default=128,
        help="Sliding window size.")
    parser.add_argument(
        "--stride",
        type=int,
        default=126,
        help="Sliding window stride.")
    parser.add_argument(
        "--workers",
        type=int,
        default=4,
        help="Number of worker processes. Start with 4 for NAS I/O.")
    parser.add_argument(
        "--suffix",
        default=".tif",
        help="Input image suffix, e.g. .tif or .tiff.")
    parser.add_argument(
        "--overwrite",
        action="store_true",
        help="Overwrite existing slice files. By default existing files are skipped.")
    parser.add_argument(
        "--keep-compression",
        action="store_true",
        help="Keep source compression settings. Default removes compression for speed.")
    parser.add_argument(
        "--gdal-cachemax",
        type=int,
        default=128,
        help="GDAL cache size in MB for each worker.")
    parser.add_argument(
        "--limit",
        type=int,
        default=None,
        help="Only process the first N base images. Useful for speed tests.")
    return parser.parse_args()


def make_starts(size, win_size, stride):
    if size < win_size:
        return []
    starts = list(range(0, size - win_size + 1, stride))
    last_start = size - win_size
    if starts[-1] != last_start:
        starts.append(last_start)
    return starts


def make_output_profile(src, window, win_size, keep_compression):
    profile = src.profile.copy()
    profile.update(
        height=win_size,
        width=win_size,
        transform=src.window_transform(window),
        tiled=False,
    )

    # Large source tiles can be invalid or slow for 128x128 patches.
    profile.pop("blockxsize", None)
    profile.pop("blockysize", None)

    if not keep_compression:
        for key in (
                "compress",
                "compression",
                "predictor",
                "zlevel",
                "jpeg_quality",
                "webp_level",
                "BIGTIFF",
        ):
            profile.pop(key, None)
    return profile


def slice_one_name(tif_name, data_root, input_folders, output_folders, win_size,
                   stride, overwrite, keep_compression, gdal_cachemax):
    try:
        import rasterio
        from rasterio.windows import Window
    except ImportError as exc:
        raise ImportError(
            "This script needs rasterio to read and write tif files. "
            "Install rasterio in the Python environment used for slicing."
        ) from exc

    in_paths = [os.path.join(data_root, folder, tif_name) for folder in input_folders]
    missing = [path for path in in_paths if not os.path.isfile(path)]
    if missing:
        return {
            "file": tif_name,
            "ok": False,
            "error": "missing input files: " + "; ".join(missing),
            "patches": 0,
            "written": 0,
            "skipped": 0,
        }

    out_dirs = [os.path.join(data_root, folder) for folder in output_folders]
    for out_dir in out_dirs:
        os.makedirs(out_dir, exist_ok=True)

    with rasterio.Env(GDAL_CACHEMAX=gdal_cachemax):
        with ExitStack() as stack:
            srcs = [stack.enter_context(rasterio.open(path)) for path in in_paths]
            ref_height, ref_width = srcs[0].height, srcs[0].width
            mismatched = [
                path for path, src in zip(in_paths, srcs)
                if src.height != ref_height or src.width != ref_width
            ]
            if mismatched:
                return {
                    "file": tif_name,
                    "ok": False,
                    "error": "input sizes do not match: " + "; ".join(mismatched),
                    "patches": 0,
                    "written": 0,
                    "skipped": 0,
                }

            row_starts = make_starts(ref_height, win_size, stride)
            col_starts = make_starts(ref_width, win_size, stride)
            if not row_starts or not col_starts:
                return {
                    "file": tif_name,
                    "ok": False,
                    "error": (
                        f"image is smaller than win_size: "
                        f"{ref_width}x{ref_height}, win_size={win_size}"),
                    "patches": 0,
                    "written": 0,
                    "skipped": 0,
                }

            file_stem = os.path.splitext(os.path.basename(tif_name))[0]
            patch_count = 0
            written_count = 0
            skipped_count = 0

            for row_off in row_starts:
                for col_off in col_starts:
                    window = Window(col_off, row_off, win_size, win_size)
                    out_name = f"{file_stem}_x{col_off}_y{row_off}.tif"
                    out_paths = [os.path.join(out_dir, out_name) for out_dir in out_dirs]

                    if not overwrite and all(os.path.exists(path) for path in out_paths):
                        patch_count += 1
                        skipped_count += len(out_paths)
                        continue

                    for src, out_path in zip(srcs, out_paths):
                        if not overwrite and os.path.exists(out_path):
                            skipped_count += 1
                            continue

                        win_data = src.read(window=window)
                        out_profile = make_output_profile(
                            src, window, win_size, keep_compression)
                        with rasterio.open(out_path, "w", **out_profile) as dst:
                            dst.write(win_data)
                        written_count += 1

                    patch_count += 1

    return {
        "file": tif_name,
        "ok": True,
        "error": "",
        "patches": patch_count,
        "written": written_count,
        "skipped": skipped_count,
    }


def main():
    args = parse_args()
    if len(args.input_folders) != len(args.output_folders):
        raise ValueError("--input-folders and --output-folders must have the same length.")

    for folder in args.output_folders:
        os.makedirs(os.path.join(args.data_root, folder), exist_ok=True)

    base_dir = os.path.join(args.data_root, args.input_folders[0])
    suffix = args.suffix.lower()
    tif_list = sorted(
        name for name in os.listdir(base_dir)
        if name.lower().endswith(suffix)
    )
    if args.limit is not None:
        tif_list = tif_list[:args.limit]

    print(f"data_root: {args.data_root}")
    print(f"input_folders: {args.input_folders}")
    print(f"output_folders: {args.output_folders}")
    print(f"images: {len(tif_list)}, workers: {args.workers}")
    print(f"win_size: {args.win_size}, stride: {args.stride}")
    print(f"overwrite: {args.overwrite}, keep_compression: {args.keep_compression}")

    total_patches = 0
    total_written = 0
    total_skipped = 0
    errors = []

    with ProcessPoolExecutor(max_workers=args.workers) as executor:
        futures = [
            executor.submit(
                slice_one_name,
                tif_name,
                args.data_root,
                args.input_folders,
                args.output_folders,
                args.win_size,
                args.stride,
                args.overwrite,
                args.keep_compression,
                args.gdal_cachemax,
            )
            for tif_name in tif_list
        ]

        for future in tqdm(as_completed(futures),
                           total=len(futures),
                           desc="Slicing images"):
            try:
                result = future.result()
            except Exception as exc:
                errors.append(f"worker exception: {exc!r}")
                continue

            if not result["ok"]:
                errors.append(f"{result['file']}: {result['error']}")
                continue

            total_patches += result["patches"]
            total_written += result["written"]
            total_skipped += result["skipped"]

    print("Done.")
    print(f"patch windows: {total_patches}")
    print(f"files written: {total_written}")
    print(f"files skipped: {total_skipped}")
    if errors:
        print(f"errors: {len(errors)}")
        for error in errors[:20]:
            print(error)
        if len(errors) > 20:
            print(f"... {len(errors) - 20} more errors not shown")


if __name__ == "__main__":
    main()
