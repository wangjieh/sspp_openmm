# calc_normalization.py
import argparse
from pathlib import Path

import numpy as np
import rasterio
from tqdm import tqdm


IMAGE_EXTS = {'.jpg', '.jpeg', '.png', '.tif', '.tiff'}


def main():
    parser = argparse.ArgumentParser(description='计算图像归一化参数')
    parser.add_argument('--img-dir', type=Path, required=True)
    args = parser.parse_args()

    paths = [
        p for p in args.img_dir.rglob('*')
        if p.is_file() and p.suffix.lower() in IMAGE_EXTS
    ]
    if not paths:
        raise FileNotFoundError('未找到 jpg/png/tif/tiff 图像。')

    total = total_sq = count = None

    for path in tqdm(paths, desc='统计图像'):
        with rasterio.open(path) as src:
            image = src.read(masked=True)  # [C, H, W]，自动掩膜 nodata

        if total is None:
            channels = image.shape[0]
            total = np.zeros(channels, dtype=np.float64)
            total_sq = np.zeros(channels, dtype=np.float64)
            count = np.zeros(channels, dtype=np.int64)
        elif image.shape[0] != len(total):
            raise ValueError(
                f'通道数不一致：{path} 为 {image.shape[0]} 通道，'
                f'预期为 {len(total)} 通道。')

        for c in range(image.shape[0]):
            values = image[c].compressed().astype(np.float64)
            total[c] += values.sum()
            total_sq[c] += np.square(values).sum()
            count[c] += values.size

    mean = total / count
    var = np.maximum(total_sq / count - mean ** 2, 0)
    std = np.sqrt(var)

    print('\n统计完成')
    print(f'图像数量: {len(paths)}')
    print(f'通道数: {len(mean)}')
    print(f'mean: {mean.tolist()}')
    print(f'var:  {var.tolist()}')
    print(f'std:  {std.tolist()}')


if __name__ == '__main__':
    main()
