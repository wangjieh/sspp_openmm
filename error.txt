import os
import tempfile
import zipfile
from pathlib import Path

import numpy as np
import torch
import tqdm

from configs.configs import cfg
from dataset.ev_uav import EvUAV
from model.evspsegnet import evspsegnet


def get_submission_zip_path(configs):
    save_path = getattr(configs, "save_path", None)
    if not save_path:
        raise ValueError(
            "TEST.save is True, but TEST.save_path is empty. "
            "Set it to a final ZIP path, for example "
            "/path/to/track2_submission.zip."
        )

    zip_path = Path(save_path).expanduser().resolve()
    if zip_path.suffix.lower() != ".zip":
        raise ValueError(f"TEST.save_path must end with .zip: {zip_path}")
    return zip_path


def load_raw_events(npz_path):
    """读取赛道 2 提交所需的原始 x/y/t/p。"""
    with np.load(npz_path, allow_pickle=False) as events:
        if "ev" not in events:
            raise KeyError(f"{npz_path} does not contain the required 'ev' field.")
        raw_events = events["ev"]

    required_fields = {"x", "y", "t", "p"}
    if raw_events.dtype.names is None or not required_fields.issubset(raw_events.dtype.names):
        raise ValueError(
            f"{npz_path} has invalid ev fields: {raw_events.dtype.names}; "
            f"expected at least {sorted(required_fields)}."
        )
    return raw_events


def write_submission_txt(raw_events, event_scores, txt_path, threshold):
    """
    每行：x y t p label
    无表头；保留原始事件顺序、原始浮点时间戳和原始极性。
    """
    event_scores = np.asarray(event_scores).reshape(-1)

    if len(raw_events) != len(event_scores):
        raise ValueError(
            f"Prediction/event count mismatch for {txt_path}: "
            f"{len(event_scores)} predictions, {len(raw_events)} raw events."
        )

    labels = (event_scores >= threshold).astype(np.int8)

    rows = np.column_stack(
        (
            raw_events["x"],
            raw_events["y"],
            raw_events["t"],
            raw_events["p"],
            labels,
        )
    )

    np.savetxt(
        txt_path,
        rows,
        fmt=("%d", "%d", "%.10f", "%d", "%d"),
        delimiter=" ",
    )


def main():
    if not cfg.save:
        raise ValueError("TEST.save is False. Set TEST.save: True before exporting.")

    if cfg.batch_size != 1:
        raise ValueError("Track 2 submission export requires batch_size: 1.")

    zip_path = get_submission_zip_path(cfg)
    if zip_path.exists():
        raise FileExistsError(
            f"Submission ZIP already exists: {zip_path}\n"
            "Please change TEST.save_path or remove the old ZIP first."
        )
    zip_path.parent.mkdir(parents=True, exist_ok=True)

    threshold = float(getattr(cfg, "submit_thresh", 0.9))
    if not 0.0 <= threshold <= 1.0:
        raise ValueError(f"TEST.submit_thresh must be in [0, 1], got {threshold}")

    net = evspsegnet(cfg).eval().cuda()
    net.load_state_dict(torch.load(cfg.model_path))
    print("dict load:", cfg.model_path)
    print("submission threshold:", threshold)

    dataset = EvUAV(cfg, mode="test")
    test_dataloader = torch.utils.data.DataLoader(
        dataset,
        batch_size=cfg.batch_size,
        collate_fn=dataset.custom_collate,
    )

    written_txt_names = []

    with tempfile.TemporaryDirectory(
        prefix="track2_submission_",
        dir=zip_path.parent,
    ) as temp_dir:
        pbar = tqdm.tqdm(
            total=len(test_dataloader),
            desc="video",
            unit="video",
            unit_scale=True,
        )

        try:
            for sample, batch in enumerate(test_dataloader):
                with torch.no_grad():
                    voxel_predictions, _ = net(batch["voxel_ev"])
                    event_scores = voxel_predictions[
                        batch["p2v_map"].long().cuda()
                    ].reshape(-1).cpu().numpy()

                npz_name = dataset.file_list[sample]
                npz_path = os.path.join(dataset.root, npz_name)
                raw_events = load_raw_events(npz_path)

                txt_name = Path(npz_name).with_suffix(".txt").name
                txt_path = os.path.join(temp_dir, txt_name)

                write_submission_txt(
                    raw_events=raw_events,
                    event_scores=event_scores,
                    txt_path=txt_path,
                    threshold=threshold,
                )

                written_txt_names.append(txt_name)
                pbar.update(1)
        finally:
            pbar.close()

        # ZIP 内只有 txt 文件，不含外层目录。
        with zipfile.ZipFile(
            zip_path,
            "w",
            compression=zipfile.ZIP_DEFLATED,
        ) as archive:
            for txt_name in sorted(written_txt_names):
                archive.write(
                    os.path.join(temp_dir, txt_name),
                    arcname=txt_name,
                )

    # 最终结构校验。
    with zipfile.ZipFile(zip_path) as archive:
        archive_names = archive.namelist()

    if sorted(archive_names) != sorted(written_txt_names):
        raise RuntimeError(f"Submission ZIP verification failed: {zip_path}")

    print("submission ZIP saved:", zip_path)
    print("TXT files:", len(written_txt_names))


if __name__ == "__main__":
    main()
