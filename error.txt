#!/usr/bin/env python
"""Convert a DINOv3/ByteDance Swin-Huge checkpoint to MMSeg 1.2.2 Swin keys."""

# python "...\convert_dinov3_swin_to_mmseg.py" input.pth output.pth --strip-prefix module --strip-prefix backbone

from __future__ import annotations

import argparse
import inspect
import sys
from collections import OrderedDict
from pathlib import Path
from typing import Any, Iterable


DEFAULT_DROP_PREFIXES = (
    "head.",
    "proj_cls.",
    "cls_norm.",
    "local_cls_norm.",
)
DEFAULT_DROP_KEYS = ("masked_embed",)


def build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(
        description=(
            "Convert DINOv3/ByteDance Swin checkpoint keys to MMSegmentation "
            "1.2.2 SwinTransformer backbone keys."
        )
    )
    parser.add_argument("src", help="Input .pt/.pth checkpoint.")
    parser.add_argument("dst", help="Output .pth checkpoint for MMSeg.")
    parser.add_argument(
        "--state-dict-key",
        default="auto",
        help=(
            "Top-level key containing the state_dict. Use 'auto' to try "
            "state_dict/model/teacher/student and then the whole checkpoint."
        ),
    )
    parser.add_argument(
        "--strip-prefix",
        action="append",
        default=[],
        help=(
            "Prefix to strip from source keys before conversion. Can be passed "
            "multiple times, for example --strip-prefix module. "
            "--strip-prefix backbone."
        ),
    )
    parser.add_argument(
        "--map-final-norm",
        default="norm3",
        help=(
            "Map source final norm.weight/bias to this MMSeg key prefix. "
            "Use 'drop' to discard source norm.*. Default: norm3."
        ),
    )
    parser.add_argument(
        "--wrap-state-dict",
        action="store_true",
        help="Save as {'state_dict': converted} instead of a bare state_dict.",
    )
    parser.add_argument(
        "--unsafe-load",
        action="store_true",
        help=(
            "Use torch.load(..., weights_only=False) when supported. "
            "Only use for checkpoints you trust."
        ),
    )
    return parser


def load_torch():
    try:
        import torch  # type: ignore
    except ImportError as exc:
        raise RuntimeError(
            "PyTorch is required to read/write .pt/.pth checkpoints. "
            "Install it with: pip install torch"
        ) from exc
    return torch


def torch_load(path: Path, unsafe_load: bool) -> Any:
    torch = load_torch()
    kwargs: dict[str, Any] = {"map_location": "cpu"}
    if "weights_only" in inspect.signature(torch.load).parameters:
        kwargs["weights_only"] = not unsafe_load

    try:
        return torch.load(path, **kwargs)
    except Exception as exc:
        if kwargs.get("weights_only") is True and not unsafe_load:
            raise RuntimeError(
                "Could not load with weights_only=True. If this checkpoint is "
                "from a trusted source, rerun with --unsafe-load."
            ) from exc
        raise


def extract_state_dict(checkpoint: Any, state_dict_key: str) -> Any:
    if state_dict_key != "auto":
        if not isinstance(checkpoint, dict):
            raise TypeError("--state-dict-key requires a dict checkpoint.")
        if state_dict_key not in checkpoint:
            available = ", ".join(str(key) for key in checkpoint.keys())
            raise KeyError(
                f"Top-level key {state_dict_key!r} not found. Available keys: {available}"
            )
        return checkpoint[state_dict_key]

    if not isinstance(checkpoint, dict):
        return checkpoint

    for key in ("state_dict", "model", "teacher", "student"):
        value = checkpoint.get(key)
        if isinstance(value, dict):
            return value

    return checkpoint


def normalize_prefix(prefix: str) -> str:
    return prefix if prefix.endswith(".") else f"{prefix}."


def strip_source_prefixes(key: str, prefixes: Iterable[str]) -> str:
    for prefix in prefixes:
        normalized = normalize_prefix(prefix)
        if key.startswith(normalized):
            return key[len(normalized) :]
    return key


def correct_unfold_reduction_order(x: Any) -> Any:
    out_channel, in_channel = x.shape
    x = x.reshape(out_channel, 4, in_channel // 4)
    x = swap_axes(x[:, [0, 2, 1, 3], :], 1, 2).reshape(out_channel, in_channel)
    return x


def correct_unfold_norm_order(x: Any) -> Any:
    in_channel = x.shape[0]
    x = x.reshape(4, in_channel // 4)
    x = swap_axes(x[[0, 2, 1, 3], :], 0, 1).reshape(in_channel)
    return x


def swap_axes(x: Any, dim0: int, dim1: int) -> Any:
    if hasattr(x, "swapaxes"):
        return x.swapaxes(dim0, dim1)
    return x.transpose(dim0, dim1)


def should_drop_key(key: str) -> bool:
    return key in DEFAULT_DROP_KEYS or key.startswith(DEFAULT_DROP_PREFIXES)


def convert_key_and_value(
    key: str,
    value: Any,
    *,
    map_final_norm: str,
) -> tuple[str | None, Any]:
    if should_drop_key(key):
        return None, value

    if key.startswith("norm."):
        if map_final_norm == "drop":
            return None, value
        suffix = key[len("norm.") :]
        return f"{map_final_norm}.{suffix}", value

    new_key = key
    new_value = value

    if key.startswith("layers."):
        if "attn." in new_key:
            new_key = new_key.replace("attn.", "attn.w_msa.")
        elif "mlp.fc1." in new_key:
            new_key = new_key.replace("mlp.fc1.", "ffn.layers.0.0.")
        elif "mlp.fc2." in new_key:
            new_key = new_key.replace("mlp.fc2.", "ffn.layers.1.")
        elif "mlp." in new_key:
            new_key = new_key.replace("mlp.", "ffn.")

        if "downsample.reduction." in new_key:
            new_value = correct_unfold_reduction_order(new_value)
        elif "downsample.norm." in new_key:
            new_value = correct_unfold_norm_order(new_value)

        new_key = new_key.replace("layers", "stages", 1)

    elif key.startswith("patch_embed."):
        new_key = new_key.replace("proj", "projection", 1)

    return new_key, new_value


def convert_state_dict(
    state_dict: dict[str, Any],
    *,
    strip_prefixes: Iterable[str] = (),
    map_final_norm: str = "norm3",
) -> tuple[OrderedDict[str, Any], dict[str, int]]:
    converted = OrderedDict()
    report = {"input": 0, "converted": 0, "skipped": 0}

    for raw_key, value in state_dict.items():
        report["input"] += 1
        key = strip_source_prefixes(str(raw_key), strip_prefixes)
        new_key, new_value = convert_key_and_value(
            key, value, map_final_norm=map_final_norm
        )
        if new_key is None:
            report["skipped"] += 1
            continue
        converted[new_key] = new_value
        report["converted"] += 1

    return converted, report


def main(argv: list[str] | None = None) -> int:
    parser = build_parser()
    args = parser.parse_args(argv)
    src = Path(args.src)
    dst = Path(args.dst)

    try:
        checkpoint = torch_load(src, unsafe_load=args.unsafe_load)
        state_dict = extract_state_dict(checkpoint, args.state_dict_key)
        if not isinstance(state_dict, dict):
            raise TypeError("The extracted state_dict is not a dict-like object.")

        converted, report = convert_state_dict(
            state_dict,
            strip_prefixes=args.strip_prefix,
            map_final_norm=args.map_final_norm,
        )

        torch = load_torch()
        dst.parent.mkdir(parents=True, exist_ok=True)
        output = {"state_dict": converted} if args.wrap_state_dict else converted
        torch.save(output, dst)
    except Exception as exc:
        print(f"Error: {exc}", file=sys.stderr)
        return 1

    print(
        f"Converted {report['converted']} key(s), skipped {report['skipped']} "
        f"key(s), wrote {dst}"
    )
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
