# calc_mean_std.py
import argparse
from pathlib import Path

import numpy as np
import rasterio
from tqdm import tqdm


IMAGE_EXTS = {'.tif', '.tiff', '.png', '.jpg', '.jpeg', '.bmp'}


def main():
    parser = argparse.ArgumentParser(
        description='使用 rasterio 计算图像数据集的逐通道均值与方差')
    parser.add_argument('--img-dir', type=Path, required=True)
    parser.add_argument('--recursive', action='store_true',
                        help='递归读取子目录')
    args = parser.parse_args()

    iterator = args.img_dir.rglob('*') if args.recursive else args.img_dir.glob('*')
    paths = [p for p in iterator
             if p.is_file() and p.suffix.lower() in IMAGE_EXTS]
    if not paths:
        raise FileNotFoundError('未找到支持的图像文件。')

    pixel_sum = pixel_sq_sum = pixel_count = None
    skipped = []

    for path in tqdm(paths, desc='统计图像'):
        try:
            with rasterio.open(path) as src:
                image = src.read(masked=True).astype(np.float64)  # [C, H, W]

            if pixel_sum is None:
                pixel_sum = np.zeros(image.shape[0], dtype=np.float64)
                pixel_sq_sum = np.zeros(image.shape[0], dtype=np.float64)
                pixel_count = np.zeros(image.shape[0], dtype=np.int64)

            if image.shape[0] != len(pixel_sum):
                raise ValueError(f'波段数不一致：{image.shape[0]} != {len(pixel_sum)}')

            for band in range(image.shape[0]):
                values = image[band].compressed()  # 自动忽略 nodata / 掩膜像素
                pixel_sum[band] += values.sum()
                pixel_sq_sum[band] += np.square(values).sum()
                pixel_count[band] += values.size

        except Exception as e:
            skipped.append(f'{path}: {e}')

    if np.any(pixel_count == 0):
        raise RuntimeError('至少有一个波段没有有效像素。')

    mean = pixel_sum / pixel_count
    variance = np.maximum(pixel_sq_sum / pixel_count - mean ** 2, 0)
    std = np.sqrt(variance)

    print('\n统计完成')
    print(f'成功图像数：{len(paths) - len(skipped)} / {len(paths)}')
    print(f'逐通道均值 mean：{mean.tolist()}')
    print(f'逐通道方差 var： {variance.tolist()}')
    print(f'逐通道标准差 std：{std.tolist()}')

    if skipped:
        print(f'\n跳过文件数：{len(skipped)}')
        for message in skipped[:10]:
            print(message)


if __name__ == '__main__':
    main()
